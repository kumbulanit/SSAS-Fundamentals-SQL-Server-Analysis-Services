SELECT
    {
        -- PRODUCTION METRICS
        [Measures].[Tonnes Produced],
        [Measures].[Average Grade],
        [Measures].[Revenue ZAR],
        [Measures].[Average Cost Per Tonne],
        
        -- OPERATING COST BREAKDOWN
        [Measures].[Labor Cost],
        [Measures].[Equipment Cost],
        [Measures].[Maintenance Cost],
        [Measures].[Safety Cost],
        [Measures].[Utilities Cost],
        [Measures].[Other Cost]
    } ON COLUMNS,
    {
        -- MINE ANALYSIS
        -- Beeshoek: ~3.7M tonnes, 66% grade, ~1.83B ZAR revenue
        -- Khumani: ~5.5M tonnes, 63% grade, ~2.82B ZAR revenue
        -- Black Rock: ~2.0M tonnes, 49% grade, ~1.22B ZAR revenue
        -- Dwarsrivier: ~1.3M tonnes, 46% grade, ~974M ZAR revenue
        -- Machadodorp: ~1.0M tonnes, 53% grade, ~975M ZAR revenue
        [Mine].[All Mines],
        [Mine].[Mine Type].[Iron Ore],
        [Mine].[Mine Type].[Manganese],
        [Mine].[Mine Type].[Chrome],
        [Mine].[Province].[Northern Cape],
        [Mine].[Province].[Limpopo],
        [Mine].[Province].[Mpumalanga],
        
        -- TEMPORAL ANALYSIS
        -- 2023: Full year data (12 months)
        -- 2024: YTD partial (Jan-Mar only)
        [Date].[Calendar Year].[2023],
        [Date].[Calendar Year].[2024],
        [Date].[Calendar Year].[2023].[Calendar Quarter].[Q1],
        [Date].[Calendar Year].[2023].[Calendar Quarter].[Q2],
        [Date].[Calendar Year].[2023].[Calendar Quarter].[Q3],
        [Date].[Calendar Year].[2023].[Calendar Quarter].[Q4],
        
        -- DEPARTMENT COST DISTRIBUTION
        -- Mining Operations: ~70% of total costs
        -- Engineering: ~15% of total costs
        -- Safety: ~5% of total costs
        [Department].[Department Name].[Mining Operations],
        [Department].[Department Name].[Engineering & Maintenance],
        [Department].[Department Name].[Safety & Compliance]
    } ON ROWS
FROM [Assmang Mining Analytics]
WHERE [Date].[Calendar Year].[2023] OR [Date].[Calendar Year].[2024];
