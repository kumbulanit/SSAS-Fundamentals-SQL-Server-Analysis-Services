SELECT
    {
        -- EQUIPMENT EFFICIENCY METRICS
        -- Expected: Equipment uptime 87-96%, downtime impacts maintenance scheduling
        [Measures].[Average UpTime Percentage],
        [Measures].[Total Downtime Minutes],
        [Measures].[Average Productivity Tonnes Per Hour],
        [Measures].[Total Maintenance Events],
        [Measures].[Total Safety Incidents],
        
        -- SAFETY KPI METRICS
        -- Expected: LTI=0, TRI 1-4/month, Compliance 96-99%, Training hours 300-600/month
        [Measures].[Total Lost Time Injuries],
        [Measures].[Total Recordable Incidents],
        [Measures].[Total Nearmisses],
        [Measures].[Average Days Without Incident],
        [Measures].[Average Compliance Score],
        [Measures].[Total Training Hours Delivered],
        
        -- EMPLOYEE METRICS
        -- Expected: Attendance 95-99%, Performance ratings 4.1-4.9/5.0, OT 2-25 hrs/month
        [Measures].[Average Attendance Percentage],
        [Measures].[Total Overtime Hours],
        [Measures].[Total Training Hours Completed],
        [Measures].[Average Performance Rating]
    } ON COLUMNS,
    {
        -- MINE-BASED ANALYSIS
        -- All 5 mines with equipment, safety, and HR metrics
        [Mine].[All Mines],
        [Mine].[Mine Name].[Beeshoek Mine],
        [Mine].[Mine Name].[Khumani Mine],
        [Mine].[Mine Name].[Black Rock Mine],
        [Mine].[Mine Name].[Dwarsrivier Chrome Mine],
        [Mine].[Mine Name].[Machadodorp Works],
        
        -- DEPARTMENT-BASED ANALYSIS
        -- Cost distribution and performance by department
        [Department].[All Departments],
        [Department].[Department Name].[Mining Operations],
        [Department].[Department Name].[Engineering & Maintenance],
        [Department].[Department Name].[Human Resources],
        [Department].[Department Name].[Safety & Compliance],
        [Department].[Department Name].[Finance & Accounting],
        [Department].[Department Name].[Corporate Affairs],
        
        -- TEMPORAL ANALYSIS
        -- Full 2023 data + YTD 2024
        [Date].[Calendar Year].[2023],
        [Date].[Calendar Year].[2024]
    } ON ROWS
FROM [Assmang Mining Analytics]
WHERE [Date].[Calendar Year].[2023] OR [Date].[Calendar Year].[2024];
