SELECT
    {
        [Measures].[Mine Count],
        [Measures].[Department Count],
        [Measures].[Employee Count],
        [Measures].[Date Count]
    } ON COLUMNS,
    {
        -- MINE DIMENSION VALIDATION
        -- Expected: 5 mines (Beeshoek, Khumani, Black Rock, Dwarsrivier, Machadodorp)
        [Mine].[All Mines],
        [Mine].[Mine Type].[Iron Ore],
        [Mine].[Mine Type].[Manganese],
        [Mine].[Mine Type].[Chrome],
        [Mine].[Province].[Northern Cape],
        [Mine].[Province].[Limpopo],
        [Mine].[Province].[Mpumalanga],
        
        -- DEPARTMENT DIMENSION VALIDATION
        -- Expected: 6 departments across 2 function areas represented
        [Department].[All Departments],
        [Department].[Function Area].[Mining Operations],
        [Department].[Function Area].[Engineering],
        [Department].[Function Area].[HR],
        [Department].[Function Area].[Safety],
        [Department].[Function Area].[Finance],
        [Department].[Function Area].[Corporate],
        
        -- EMPLOYEE DIMENSION VALIDATION
        -- Expected: 18 employees distributed across departments
        [Employee].[All Employees],
        
        -- DATE DIMENSION VALIDATION
        -- Expected: 730 days (2023-01-01 to 2024-12-31)
        [Date].[Calendar Year].[2023],
        [Date].[Calendar Year].[2024]
    } ON ROWS
FROM [Assmang Mining Analytics];
